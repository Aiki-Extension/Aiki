/** Start the tour */

/** Restart tour link */
if (tour.ended()) {
    chrome.tabs.getSelected(null, tab => {
        if (tab.url.indexOf('/assets/html/introTour.html') !== -1) {
            tour.restart();
        }
    });
}

/** get current tab */
chrome.tabs.getSelected(null, tab =>{
    id = tab.id;
});

/** end tour if tab closed */
chrome.tabs.onRemoved.addListener(tabId => {
    if (tabId === id) {
        tour.end();
    }
});
